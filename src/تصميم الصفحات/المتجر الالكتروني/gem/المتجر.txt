// ✅✅✅ تعريف مكون AnimatedFeatureCard هنا ✅✅✅
interface AnimatedFeatureCardProps {
  icon: React.ReactNode;
  title: string;
  description: string;
  delay?: number;
  locale: string | string[] | undefined;
  linkPath: string; // Made non-optional as it's used in Link
}

const AnimatedFeatureCard: React.FC<AnimatedFeatureCardProps> = ({ icon, title, description, delay = 0, locale, linkPath }) => {
  const controls = useAnimation();
  const [ref, inView] = useInView({ triggerOnce: true, threshold: 0.2 });

  useEffect(() => {
    if (inView) {
      controls.start("visible");
    }
  }, [controls, inView]);

  const cardVariantsLocal = {
    hidden: { opacity: 0, y: 50, scale: 0.9 },
    visible: { opacity: 1, y: 0, scale: 1, transition: { duration: 0.6, delay, ease: [0.6, -0.05, 0.01, 0.99] } },
  };

  let finalLinkPath = linkPath;
  if (typeof locale === 'string' && locale) {
    const pathSegment = linkPath.startsWith('/') ? linkPath : `/${linkPath}`;
    finalLinkPath = `/${locale}${pathSegment}`;
  } else if (linkPath && !linkPath.startsWith('/')) {
     finalLinkPath = `/${linkPath}`;
  }


  return (
    <motion.div
      ref={ref}
      variants={cardVariantsLocal}
      initial="hidden"
      animate={controls}
      className="bg-topshield-white p-8 rounded-2xl shadow-topshield-lg text-center flex flex-col items-center h-full border border-topshield-border-subtle/50"
    >
      <motion.div
        className="text-5xl text-topshield-accent-gold mb-6 p-4 bg-topshield-accent-gold/10 rounded-full"
        whileHover={{ scale: 1.1, rotate: 5 }}
      >
        {icon}
      </motion.div>
      <h3 className="text-2xl font-bold font-heading text-topshield-primary-navy mb-3">{title}</h3>
      <p className="text-topshield-text-medium leading-relaxed flex-grow">{description}</p>
      <Link href={finalLinkPath} legacyBehavior>
        <a className="mt-6 inline-flex items-center text-topshield-primary-blue hover:text-topshield-accent-gold font-semibold group">
          اعرف المزيد <FiArrowRight className="mr-2 transition-transform duration-300 group-hover:translate-x-1" />
        </a>
      </Link>
    </motion.div>
  );
};

const ProductDisplayCard: React.FC<ProductDisplayCardProps> = ({ product, locale, onAddToCart }) => {

        {product.productLine && <span className="text-xs font-semibold text-topshield-accent-gold-darker mb-1">{product.productLine} {product.vltGrade && `- ${product.vltGrade}%`}</span>}
        <h3 className="text-lg font-bold font-heading text-topshield-primary-navy mb-1.5 group-hover:text-topshield-accent-gold-darker transition-colors line-clamp-2 leading-tight">
          <Link href={detailLink} passHref legacyBehavior><a>{product.name}</a></Link>
        </h3>
        {product.rating && (
          <div className="flex items-center text-sm text-amber-500 mb-2">
            {[...Array(Math.floor(product.rating))].map((_, i) => <FaStar key={`star-${i}`} />)}
            {product.rating % 1 !== 0 && <FaStar key="half-star" style={{ clipPath: 'inset(0 50% 0 0)' }} />}
            {[...Array(5 - Math.ceil(product.rating))].map((_, i) => <FiStarOutline key={`empty-${i}`} className="text-gray-300" />)}

          </div>
        )}
        <p className="text-sm text-topshield-text-medium mb-3 line-clamp-2 flex-grow">{product.shortDescription}</p>
        
        <div className="flex items-end justify-between mb-4">
          <div>
            <p className="text-2xl font-bold text-topshield-primary-blue">
              {product.price.toLocaleString()} <span className="text-sm font-normal">{product.currency}</span>
            </p>
            {product.originalPrice && (
              <p className="text-sm text-topshield-text-light line-through">
                {product.originalPrice.toLocaleString()} {product.currency}
              </p>
            )}
          </div>
        </div>

        <div className="mt-auto">
          <motion.button
            onClick={handleAddToCart}
            className={`w-full font-semibold py-3 px-4 rounded-lg transition-all duration-200 ease-in-out shadow-md hover:shadow-lg focus:outline-none focus:ring-2 focus:ring-offset-2 flex items-center justify-center gap-2
              ${addedToCart 
                ? 'bg-green-500 text-white cursor-default' 
                : 'bg-topshield-accent-gold hover:bg-topshield-accent-gold-darker text-topshield-primary-navy focus:ring-topshield-accent-gold-darker'
              }`}
            whileHover={!addedToCart ? { scale: 1.03, y: -1 } : {}}
            whileTap={!addedToCart ? { scale: 0.97 } : {}}
            disabled={addedToCart}
          >
            {addedToCart ? <FiCheckCircle className="text-xl"/> : <FiShoppingCart className="text-xl"/>}
            {addedToCart ? 'تمت الإضافة للسلة!' : 'أضف إلى السلة'}
          </motion.button>
        </div>
      </div>
    </motion.div>
  );
};
